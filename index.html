How It Works

# Initialize a Canon project
canon init

# Your canon.yml defines dependencies
canon: 1.0
registry:
  default: https://spec.farm

dependencies:
  - canon-protocol.org/type@1.0.0
  - canon-protocol.org/transformation@1.0.0
  - canon-protocol.org/auto-version@1.0.0

# Install dependencies
canon install

# Create a specification from sources
canon create my-api --type api.io/openapi@2.0.0

# Transform sources into canonical form
canon transform my-api/

# Sign and publish
canon publish my-api/
